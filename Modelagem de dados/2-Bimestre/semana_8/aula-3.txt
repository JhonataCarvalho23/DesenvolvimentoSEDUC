CREATE TABLE produtos (
    id_produto INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100),
    categoria VARCHAR(100),
    preco DECIMAL(10,2)
);

-- Tabela de vendas
CREATE TABLE vendas (
    id_venda INT PRIMARY KEY AUTO_INCREMENT,
    id_produto INT,
    quantidade INT,
    data_venda DATE,
    FOREIGN KEY (id_produto) REFERENCES produtos(id_produto)
);

SELECT * FROM produtos;
SELECT * FROM vendas;

SELECT p.nome, SUM(v.quantidade) AS total
-- Esse select ocorre a aplicação do innerjoin, mostra o total de vendas nos ultimos 3 meses
FROM produtos p
JOIN vendas v ON p.id_produto = v.id_produto
WHERE v.data_venda >= DATE_SUB(CURDATE(), INTERVAL 3 MONTH)
GROUP BY p.nome
ORDER BY total DESC
LIMIT 1;

SELECT p.categoria, SUM(v.quantidade * p.preco) AS total
-- Esse select ocorre a soma total da categoria de produtos
FROM produtos p
JOIN vendas v ON p.id_produto = v.id_produto
WHERE v.data_venda BETWEEN '2026-03-01' AND '2026-03-31'
GROUP BY p.categoria;

SELECT AVG(quantidade)
-- Esse select faz a media aritimetica dos ultimos 3 meses
FROM vendas
WHERE data_venda >= DATE_SUB(CURDATE(), INTERVAL 3 MONTH);

SELECT 
-- Esse select vai mostraro produto mais caro e o produto mais barato em ordem decrescente
MAX(v.quantidade * p.preco) AS maior,
MIN(v.quantidade * p.preco) AS menor
FROM vendas v
JOIN produtos p ON p.id_produto = v.id_produto
WHERE v.data_venda >= DATE_SUB(CURDATE(), INTERVAL 3 MONTH);

SELECT p.categoria, SUM(v.quantidade) AS total
-- O ultimo select mostra em ordem decrescente a venda de categoria de produtos
FROM produtos p
JOIN vendas v ON p.id_produto = v.id_produto
WHERE v.data_venda >= DATE_SUB(CURDATE(), INTERVAL 3 MONTH)
GROUP BY p.categoria
ORDER BY total DESC
LIMIT 1;
